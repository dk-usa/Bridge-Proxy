   ```typescript
   function extractRequestText(request: AnthropicMessageRequest): string {
     // Per D-02: Embed messages array + system prompt
     const parts: string[] = [];
     if (request.system) {
       parts.push(
         typeof request.system === 'string'
           ? request.system
           : request.system.map((b) => b.text).join(' ')
       );
     }
     for (const msg of request.messages) {
       const content = msg.content;
       if (typeof content === 'string') {
         parts.push(content);
       } else {
         parts.push(content.map((b) => b.text ?? '').join(' '));
       }
     }
     return parts.join('\n');
   }
   ```

3. **checkSemanticCache Function:**

   ```typescript
   async function checkSemanticCache(
     request: AnthropicMessageRequest,
     tenantId: string
   ): Promise<{ hit: boolean; response?: AnthropicMessageResponse; embedding?: number[] }> {
     const config = getConfig();

     // D-10: Disabled by default
     if (!config.semanticCache?.enabled) {
       return { hit: false };
     }

     // D-09: Check after exact cache miss (caller responsibility)
     const text = extractRequestText(request);
     const embedding = await semanticCacheService.generateEmbedding(text);

     // D-07: Fallback on embedding failure
     if (!embedding) {
       logger.warn('Semantic cache: embedding generation failed');
       return { hit: false };
     }

     const entry = await semanticCacheService.findSimilar<AnthropicMessageResponse>(
       embedding,
       tenantId
     );

     if (entry) {
       return { hit: true, response: entry.data, embedding };
     }

     return { hit: false, embedding };
   }
   ```

4. **storeSemanticResponse Function:**

   ```typescript
   async function storeSemanticResponse(
     request: AnthropicMessageRequest,
     response: AnthropicMessageResponse,
     embedding: number[],
     tenantId: string
   ): Promise<void> {
     const config = getConfig();
     if (!config.semanticCache?.enabled) return;

     const key = semanticCacheService.generateKey(embedding);
     await semanticCacheService.set(key, embedding, response, tenantId);
   }
   ```

5. **Export Module:**
   - Export checkSemanticCache, storeSemanticResponse, extractRequestText
   - Singleton instance or factory function

Per decisions D-02, D-07, D-09, D-10.

   ```typescript
   // In POST /v1/messages handler:

   // Step 1: Exact cache check (existing)
   const cacheKey = cacheService.generateRequestKey(model, messages, params);
   const exactCached = await cacheService.get(cacheKey, tenantId);
   if (exactCached) {
     return reply.status(200).send(exactCached); // D-09: Exact hit first
   }

   // Step 2: Semantic cache check (new - D-09)
   const semanticResult = await checkSemanticCache(anthropicRequest, tenantId);
   if (semanticResult.hit && semanticResult.response) {
     // Log semantic cache hit
     logger.info({ requestId, similarity: 'semantic' }, 'Semantic cache hit');
     return reply.status(200).send(semanticResult.response);
   }

   // Step 3: Provider call (existing pipeline)
   const response = await processRequest(anthropicRequest, { requestId, tenantId });

   // Step 4: Store in caches (new + existing)
   await cacheService.set(cacheKey, response, tenantId); // Exact cache

   if (semanticResult.embedding) {
     // Only store if we generated an embedding
     await storeSemanticResponse(anthropicRequest, response, semanticResult.embedding, tenantId);
   }

   return reply.status(200).send(response);
   ```
